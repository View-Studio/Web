
< 배열 추가, 삭제 메소드 >
===============================================================================

[ unshift ]
-----------------------------------------------------------------------------------------------------------
 - 배열 앞부분에 값을 삽입 -

[ array.unshift() ]

ex)
var arr = [ 1, 2, 3, 4 ];
arr.unshift( 0 );
console.log( arr );    // [ 0, 1, 2, 3, 4 ]
-----------------------------------------------------------------------------------------------------------



[ push ]
-----------------------------------------------------------------------------------------------------------
 - 배열 뒷부분에 값을 삽입 -

[ array.push() ]

ex)
var arr = [ 1, 2, 3, 4 ];
arr.push( 5 );
console.log( arr );    // [ 1, 2, 3, 4, 5 ]
-----------------------------------------------------------------------------------------------------------



[ shift ]
-----------------------------------------------------------------------------------------------------------
 - 배열 앞부분의 값을 삭제 -

[ array.shift() ]

ex)
var arr = [ 1, 2, 3, 4 ];
arr.shift();
console.log( arr );    // [ 2, 3, 4 ]
-----------------------------------------------------------------------------------------------------------



[ pop ]
-----------------------------------------------------------------------------------------------------------
 - 배열 뒷부분의 값을 삭제 -

[ array.pop() ]

ex)
var arr = [ 1, 2, 3, 4 ];
arr.pop();
console.log( arr );    // [ 1, 2, 3 ]
-----------------------------------------------------------------------------------------------------------



[ splice ]
-----------------------------------------------------------------------------------------------------------
 - 배열의 특정위치에 요소를 추가하거나 삭제 -

[ array.splice( index, 제거할 요소 개수, 배열에 추가될 요소 ) ]

ex)
var arr = [ 1, 2, 3, 4, 5, 6, 7 ];
arr.splice( 3, 2 );
console.log( arr );    // [ 1, 2, 3, 6, 7 ]   3번째 인덱스에서부터 2개 제거 (4, 5 삭제)


var arr = [ 1, 2, 3, 4, 5, 6, 7 ];
arr.splice( 2, 1, "a", "b");
console.log( arr );    // [ 1, 2, "a", "b", 4, 5, 6, 7 ] 2번째 인덱스에서 1개 제거 후 "a"와 "b"를 추가 
-----------------------------------------------------------------------------------------------------------
