


[ toString ]
-----------------------------------------------------------------------------------------------------------
 - 배열을 문자열로 바꾸어 반환한다  -

[ array.some( func() ) ]

ex)
var arr =[ 1, 2, 3, 4 ];
console.log( arr.toString() ); // 1, 2, 3, 4
-----------------------------------------------------------------------------------------------------------



[ valueOf ]
-----------------------------------------------------------------------------------------------------------
 - toString과 비슷, 그러나 배열을 반환  -

[ array.some( func() ) ]

ex)
var arr =[ 1, 2, 3, 4 ];
console.log( arr.valueOf() ); // [ 1, 2, 3, 4 ]
-----------------------------------------------------------------------------------------------------------



[ join ]
-----------------------------------------------------------------------------------------------------------
 - 배열 원소 전부를 하나의 문자열로 합친다.  -

[ array.some( func() ) ]

ex)
var arr =[ 1, 2, 3, 4 ];
console.log( arr.join() );      // 1,2,3,4
console.log( arr.join( '-' ) ); // 1-2-3-4
-----------------------------------------------------------------------------------------------------------
